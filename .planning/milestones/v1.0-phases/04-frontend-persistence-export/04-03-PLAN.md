**Replace `saveCourseToHistory` stub** with full implementation:
```javascript
function saveCourseToHistory(course, subject, skillLevel) {
  const entry = {
    subject: subject,
    skillLevel: skillLevel,
    generatedAt: new Date().toISOString(),
    course: course
  };
  let history = loadHistory();
  history.unshift(entry);
  if (history.length > 10) history = history.slice(0, 10);
  saveHistory(history);
  renderHistory(history);
}
```

**`loadHistory()`** — read `'ylc_history'` from localStorage, parse as JSON, default to `[]`. Wrap in try/catch (corrupt data returns `[]`).

**`saveHistory(entries)`** — try/catch localStorage.setItem. On `QuotaExceededError` or `NS_ERROR_DOM_QUOTA_REACHED`: sort entries by `generatedAt` ascending, remove the first (oldest), retry once. On second failure: silent fail.
```javascript
function saveHistory(entries) {
  try {
    localStorage.setItem('ylc_history', JSON.stringify(entries));
  } catch (e) {
    if (e.name === 'QuotaExceededError' || e.name === 'NS_ERROR_DOM_QUOTA_REACHED') {
      const sorted = [...entries].sort((a, b) => new Date(a.generatedAt) - new Date(b.generatedAt));
      sorted.shift();
      try {
        localStorage.setItem('ylc_history', JSON.stringify(sorted));
      } catch (_) { /* silent fail */ }
    }
  }
}
```
