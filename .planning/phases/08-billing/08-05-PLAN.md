```javascript
// ─── Upgrade Prompt ────────────────────────────────────────────────────────
// Uses DOM construction (not innerHTML) to safely insert the upgrade link.
// upgradeUrl is server-controlled (CLERK_ACCOUNT_PORTAL_URL env var), not user input.

function showUpgradePrompt(message, upgradeUrl) {
  showContentSection('error-section');
  const el = document.getElementById('error-message');
  el.textContent = '';
  const text = document.createTextNode(message + ' ');
  const link = document.createElement('a');
  link.href = upgradeUrl || '#';
  link.textContent = 'Upgrade now';
  link.target = '_blank';
  link.rel = 'noopener noreferrer';
  el.appendChild(text);
  el.appendChild(link);
}

// ─── Usage Gate Preflight ──────────────────────────────────────────────────
// Calls GET /api/usage-check before opening EventSource.
// Native EventSource cannot read HTTP status codes — this fetch() detects 429.

async function checkUsageGate() {
  try {
    const res = await fetch('/api/usage-check');
    if (res.status === 429) {
      const body = await res.json();
      return { allowed: false, message: body.message, upgradeUrl: body.upgradeUrl };
    }
    return { allowed: true };
  } catch (_) {
    // Network error — let EventSource handle it (fail open, not fail closed)
    return { allowed: true };
  }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → GET /api/usage-check | requireUser middleware enforces auth — unauthenticated fetch() gets 401 |
| 429 upgradeUrl → browser link | URL comes from server env var; not from user input; no open redirect from user-controlled data |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-8-05-01 | Tampering | showUpgradePrompt link href | accept | upgradeUrl is server-provided (from process.env.CLERK_ACCOUNT_PORTAL_URL). Not user-controlled. Using DOM construction (createElement) rather than innerHTML — no XSS risk from server-controlled URL. |
| T-8-05-02 | Elevation of Privilege | Bypass preflight, open EventSource directly | mitigate | Frontend preflight is UX-only — gate is also enforced server-side in GET /api/course-stream (Plan 03). A user bypassing the preflight by opening EventSource directly still gets the 429 JSON from the server gate. Server gate is the authoritative control. |
| T-8-05-03 | Information Disclosure | upgradeUrl in 429 JSON | accept | CLERK_ACCOUNT_PORTAL_URL is a public URL (Clerk's hosted Account Portal — no credentials in URL). Acceptable to expose in JSON response. |
| T-8-05-04 | Denial of Service | async click handler blocking form | accept | checkUsageGate() has a try/catch that fails open (returns { allowed: true } on network error). Form is not permanently disabled. disableForm() runs synchronously before await. |
</threat_model>
