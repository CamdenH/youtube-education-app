```javascript
document.addEventListener('DOMContentLoaded', () => {
  const btn = document.getElementById('btn-connect');
  const progressFill = document.getElementById('progress-fill');
  const eventLog = document.getElementById('event-log');
  const emptyState = document.getElementById('empty-state');
  let eventSource = null;
  let startTime = null;

  btn.addEventListener('click', () => {
    // Reset state
    eventLog.querySelectorAll('.event-item').forEach(el => el.remove());
    emptyState.style.display = 'block';
    progressFill.style.width = '0%';
    progressFill.style.background = 'var(--color-accent)';
    btn.textContent = 'Connecting...';
    btn.disabled = true;
    startTime = Date.now();

    eventSource = new EventSource('/api/course-stream');

    eventSource.onopen = () => {
      btn.textContent = 'Streaming...';
      emptyState.style.display = 'none';
    };

    const EVENTS = ['query_generated', 'videos_fetched', 'scored', 'transcripts_fetched', 'course_assembled'];
    EVENTS.forEach(eventName => {
      eventSource.addEventListener(eventName, (e) => {
        const data = JSON.parse(e.data);
        addEventRow(eventName, data, eventName === 'course_assembled' ? '--color-success' : '--color-accent');
        progressFill.style.width = `${(data.step / data.total) * 100}%`;

        if (eventName === 'course_assembled') {
          progressFill.style.background = 'var(--color-success)';
          btn.textContent = 'Connect to SSE';
          btn.disabled = false;
          eventSource.close();
        }
      });
    });

    eventSource.addEventListener('error', (e) => {
      let data;
      try { data = JSON.parse(e.data); } catch { data = { message: 'Something went wrong. Check the server logs.' }; }
      addEventRow('error', data, '--color-destructive');
      progressFill.style.background = 'var(--color-destructive)';
      btn.textContent = 'Connect to SSE';
      btn.disabled = false;
      eventSource.close();
    });

    eventSource.onerror = () => {
      // Native EventSource error (connection lost, not a named 'error' event)
      btn.textContent = 'Connect to SSE';
      btn.disabled = false;
      if (eventSource) eventSource.close();
    };
  });

  function addEventRow(eventName, data, dotColor) {
    emptyState.style.display = 'none';
    const elapsed = ((Date.now() - startTime) / 1000).toFixed(1);
    const row = document.createElement('div');
    row.className = 'event-item';
    row.innerHTML = `
      <span class="dot" style="background: var(${dotColor})"></span>
      <span class="event-name">${eventName}</span>
      <span class="event-message">${data.message || ''}</span>
      <span class="event-time">${elapsed}s</span>
    `;
    eventLog.appendChild(row);
  }
});
```

This task wires the INFR-04 SSE surfacing path. In Phase 1 the stub pipeline never actually throws YouTubeQuotaError (it uses hardcoded data), but this catch path must exist so that when Phase 2 replaces the stub with real YouTube API calls, quota errors automatically surface as user-readable SSE error events.

**Modify server.js** to wrap the courseStreamHandler call with error handling. There are two approaches -- choose the one that best fits the existing code structure:

**Option A -- Wrapper middleware in server.js:**
```javascript
const { sendEvent } = require('./sse');
const { YouTubeQuotaError } = require('./youtube');

// Replace the direct route with a wrapper
app.get('/api/course-stream', async (req, res) => {
  try {
    await courseStreamHandler(req, res);
  } catch (err) {
    // If headers already sent (SSE stream started), emit error event
    if (res.headersSent) {
      if (err instanceof YouTubeQuotaError) {
        sendEvent(res, 'error', { code: 'QUOTA_EXCEEDED', message: 'YouTube quota exceeded. Try again tomorrow.' });
      } else {
        sendEvent(res, 'error', { code: 'INTERNAL', message: err.message });
      }
      res.end();
    } else {
      // Headers not sent yet -- respond with 500
      res.status(500).json({ error: err.message });
    }
  }
});
```

**Option B -- If courseStreamHandler already has internal try/catch:** Add the YouTubeQuotaError instanceof check inside sse.js's courseStreamHandler. Prefer Option A to keep error-to-SSE wiring visible in server.js.

Key details:
- Import `sendEvent` from `./sse` and `YouTubeQuotaError` from `./youtube` in server.js
- The `res.headersSent` check is critical: if the SSE headers have already been flushed, we must use `sendEvent` (not `res.status().json()`). If headers are not yet sent (error during setup), fall back to a standard JSON error response.
- After sending the error event, call `res.end()` to close the stream.
