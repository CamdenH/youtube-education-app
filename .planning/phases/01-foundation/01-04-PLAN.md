**server.js** (~30-40 lines, thin entry point):
```javascript
require('dotenv').config();
const express = require('express');
const { courseStreamHandler } = require('./sse');
const { transcriptHandler } = require('./transcript');

const app = express();
const PORT = process.env.PORT || 3000;

// Serve static files from project root (index.html)
app.use(express.static(__dirname));

// API routes
app.get('/api/course-stream', courseStreamHandler);
app.get('/api/transcript/:videoId', transcriptHandler);

// Only start listening if this file is run directly (not required by tests)
if (require.main === module) {
  app.listen(PORT, () => {
    console.log(`Server running at http://localhost:${PORT}`);
  });
}

module.exports = app; // Export for testing
```

Key points:
- `require('dotenv').config()` is the FIRST line (before any process.env access)
- `express.static(__dirname)` serves index.html at root
- `require.main === module` guard allows test imports without starting the server
- Export `app` for supertest-style testing

`<script>` block (at bottom of body) with:
```javascript
document.addEventListener('DOMContentLoaded', () => {
  const btn = document.getElementById('btn-connect');
  const progressFill = document.getElementById('progress-fill');
  const eventLog = document.getElementById('event-log');
  const emptyState = document.getElementById('empty-state');
  let eventSource = null;
  let startTime = null;

  btn.addEventListener('click', () => {
    // Reset state
    eventLog.querySelectorAll('.event-item').forEach(el => el.remove());
    emptyState.style.display = 'block';
    progressFill.style.width = '0%';
    progressFill.style.background = 'var(--color-accent)';
    btn.textContent = 'Connecting...';
    btn.disabled = true;
    startTime = Date.now();

    eventSource = new EventSource('/api/course-stream');

    eventSource.onopen = () => {
      btn.textContent = 'Streaming...';
      emptyState.style.display = 'none';
    };

    const EVENTS = ['query_generated', 'videos_fetched', 'scored', 'transcripts_fetched', 'course_assembled'];
    EVENTS.forEach(eventName => {
      eventSource.addEventListener(eventName, (e) => {
        const data = JSON.parse(e.data);
        addEventRow(eventName, data, eventName === 'course_assembled' ? '--color-success' : '--color-accent');
        progressFill.style.width = `${(data.step / data.total) * 100}%`;

        if (eventName === 'course_assembled') {
          progressFill.style.background = 'var(--color-success)';
          btn.textContent = 'Connect to SSE';
          btn.disabled = false;
          eventSource.close();
        }
      });
    });

    eventSource.addEventListener('error', (e) => {
      let data;
      try { data = JSON.parse(e.data); } catch { data = { message: 'Something went wrong. Check the server logs.' }; }
      addEventRow('error', data, '--color-destructive');
      progressFill.style.background = 'var(--color-destructive)';
      btn.textContent = 'Connect to SSE';
      btn.disabled = false;
      eventSource.close();
    });

    eventSource.onerror = () => {
      // Native EventSource error (connection lost, not a named 'error' event)
      btn.textContent = 'Connect to SSE';
      btn.disabled = false;
      if (eventSource) eventSource.close();
    };
  });

  function addEventRow(eventName, data, dotColor) {
    emptyState.style.display = 'none';
    const elapsed = ((Date.now() - startTime) / 1000).toFixed(1);
    const row = document.createElement('div');
    row.className = 'event-item';
    row.innerHTML = `
      <span class="dot" style="background: var(${dotColor})"></span>
      <span class="event-name">${eventName}</span>
      <span class="event-message">${data.message || ''}</span>
      <span class="event-time">${elapsed}s</span>
    `;
    eventLog.appendChild(row);
  }
});
```

Connection state management per UI-SPEC:
- Idle: "Connect to SSE", enabled, accent background
- Connecting: "Connecting...", disabled, opacity 0.6
- Streaming: "Streaming...", disabled, opacity 0.6
- Complete (success): "Connect to SSE", re-enabled, progress bar 100% in success color
- Complete (error): "Connect to SSE", re-enabled, progress bar in destructive color

CRITICAL per INFR-05: `eventSource.close()` is called on BOTH `course_assembled` AND `error` events to prevent auto-reconnect.

Create `tests/unit/server.test.js` using `node:test` and `node:assert`.

Tests must import `app` from `../../server.js` (the export) and start a temporary server on port 0 (random available port) for each test.

**Test cases:**

1. `GET / returns 200 and HTML content`:
   - Start server on port 0, get address
   - `fetch('http://localhost:PORT/')` — assert status 200
   - Assert `content-type` header contains `text/html`
   - Close server

2. `GET /api/course-stream returns SSE headers`:
   - Start server on port 0
   - `fetch('http://localhost:PORT/api/course-stream')` — assert status 200
   - Assert `content-type` header is `text/event-stream`
   - Assert `cache-control` header is `no-cache`
   - Abort the fetch after checking headers (don't wait for stream to complete)
   - Close server

3. `GET /api/transcript/nonexistent returns response` (verifies route is registered):
   - Start server on port 0
   - `fetch('http://localhost:PORT/api/transcript/nonexistent')` — assert status is 404 or 200 (route exists, not 404 from Express "Cannot GET")
   - Close server

4. `GET /nonexistent returns 404`:
   - Start server on port 0
   - `fetch('http://localhost:PORT/nonexistent-api-route')` — assert status is 404
   - Close server

Use `after()` hooks to ensure server is closed even on test failure. Use `AbortController` with a timeout for the SSE test to avoid hanging.

Pattern for each test:
```javascript
test('description', async () => {
  const server = app.listen(0);
  const port = server.address().port;
  try {
    // ... assertions
  } finally {
    server.close();
  }
});
```
