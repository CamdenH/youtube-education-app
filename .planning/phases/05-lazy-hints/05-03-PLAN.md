**CHANGE 2: Add hint helper functions to the `<script>` block**

In the `<script>` block, near the top of the script section (before the `renderCourse` function, after other utility functions like `slugify`, `formatDuration`, `scoreBadgeColor`, etc.), add:

```javascript
    // ── Hint localStorage helpers ─────────────────────────────────────────

    const HINTS_KEY = 'ylc_hints';

    function persistHints(videoId, hints) {
      try {
        const stored = JSON.parse(localStorage.getItem(HINTS_KEY) || '{}');
        stored[videoId] = hints;
        localStorage.setItem(HINTS_KEY, JSON.stringify(stored));
      } catch (_) { /* best-effort */ }
    }

    function loadHints(videoId) {
      try {
        const stored = JSON.parse(localStorage.getItem(HINTS_KEY) || '{}');
        return stored[videoId] || null;
      } catch (_) {
        return null;
      }
    }

    // ── Hint DOM helpers ──────────────────────────────────────────────────

    function renderHints(videoId, hints, questionsContainer) {
      const hintEls = questionsContainer.querySelectorAll('.hint-text[data-video-id="' + videoId + '"]');
      hintEls.forEach(function(el, i) {
        if (hints[i]) {
          el.innerHTML = '';
          const label = document.createElement('span');
          label.className = 'hint-label';
          label.textContent = 'Thinking point: ';
          el.appendChild(label);
          el.appendChild(document.createTextNode(hints[i]));
          el.hidden = false;
        }
      });
    }

    function showHintError(container, retryFn) {
      clearHintError(container);
      const err = document.createElement('p');
      err.className = 'hint-error';
      err.dataset.hintError = 'true';
      err.textContent = 'Could not load hints. ';
      const retryBtn = document.createElement('button');
      retryBtn.className = 'btn-retry-hint';
      retryBtn.textContent = 'Try again';
      retryBtn.addEventListener('click', retryFn);
      err.appendChild(retryBtn);
      container.appendChild(err);
    }

    function clearHintError(container) {
      const existing = container.querySelector('[data-hint-error]');
      if (existing) existing.remove();
    }

    // ── Hint fetch ────────────────────────────────────────────────────────

    async function fetchHints(video, btn, container) {
      // Build transcriptSnippet from assembled course data (first 500 words)
      const rawText = video.transcriptSnippet || '';
      const transcriptSnippet = rawText.split(/\s+/).filter(Boolean).slice(0, 500).join(' ');

      // Loading state — scoped to this button only
      btn.disabled = true;
      btn.textContent = 'Loading hints...';
      clearHintError(container);

      try {
        const res = await fetch('/api/hints', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            videoId: video.videoId,
            videoTitle: video.title,
            questions: video.questions.map(function(q) { return q.text; }),
            transcriptSnippet: transcriptSnippet,
          }),
        });

        if (!res.ok) {
          throw new Error('Server error ' + res.status);
        }

        const data = await res.json();
        renderHints(video.videoId, data.hints, container);
        persistHints(video.videoId, data.hints);
        btn.hidden = true;
      } catch (err) {
        btn.disabled = false;
        btn.textContent = 'Reveal thinking points';
        showHintError(container, function() { fetchHints(video, btn, container); });
      }
    }
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser → POST /api/hints | User-initiated fetch from index.html client-side JS to the Express server |
| index.html → localStorage | Hint data written and read from ylc_hints key in browser storage |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-03-01 | Tampering | fetchHints input construction | mitigate | The client sends `video.videoId`, `video.title`, and `video.questions.map(q => q.text)` — all sourced from the server-assembled course JSON, not from user-editable fields. transcriptSnippet is also from server data. No free-text user input enters the hint request body. |
| T-05-03-02 | Information Disclosure | hints rendered in DOM | accept | Hint text from Claude is rendered as text nodes (via `document.createTextNode`), not innerHTML — no XSS risk. The `renderHints` function uses `el.innerHTML = ''` to clear then `appendChild` with text nodes only. |
| T-05-03-03 | Tampering | ylc_hints localStorage manipulation | accept | localStorage is per-origin, client-side only. A user can manipulate their own hints; this affects only their session. No server state is affected. No PII or billing data stored in ylc_hints. |
| T-05-03-04 | Denial of Service | rapid repeated button clicks | mitigate | Button is set to `disabled = true` immediately on click and stays disabled until success (hidden) or failure (re-enabled). A user cannot fire multiple concurrent requests for the same video. |
</threat_model>
